"""Example palette code format. Files starting with "_" are ignored — copy this
file, drop the underscore from the name, edit it, and press F5 in Celpix to
load it (code plugins ask for confirmation the first time, since they run with
the app's privileges).

A code format implements a palette interpretation directly, for encodings no
built-in engine's parameters can express (table lookups, packed shares, ...).
It appears in the palette format picker like any preset. decode returns a
Palette of 0xAARRGGBB ints; encode is its exact inverse.
"""

from celpix.core.palette import Palette
from celpix.plugins import FormatInfo


class Gray4:
    """One byte per entry, but only the top 4 bits matter: a 16-step gray ramp."""

    # The id doubles as the picker entry's stable identity; the name is shown.
    info = FormatInfo(id="format.palette.example-gray4", name="Example gray (4-bit)")

    def decode(self, data, ctx):
        colors = []
        for b in data:
            v = (b >> 4) * 17  # 0..15 -> 0..255
            colors.append(0xFF000000 | (v << 16) | (v << 8) | v)
        return Palette(colors)

    def encode(self, palette, ctx):
        # Store the red channel's top nibble; decode's inverse for gray values.
        return bytes(((c >> 16) & 0xF0) for c in palette.colors)

    def bytes_per_entry(self):
        # How many bytes one entry occupies — Celpix uses this to size palette
        # reads (e.g. "load 256 entries from here").
        return 1


def register(registry):
    registry.register_format(Gray4())
