"""Example palette code format: NES/Famicom with a *custom* PPU master palette.

Files starting with "_" are ignored — copy this file, drop the underscore from
the name, edit it, and press F5 in Celpix to load it (code plugins ask for
confirmation the first time, since they run with the app's privileges).

Celpix already ships "NES/Famicom (64-color indexed)", but its 64 RGB colors are
baked in. The NES PPU doesn't store RGB at all — a palette byte is a 6-bit index
(0..63) into a *master palette* the hardware/emulator decides, and everyone has a
favourite (Bisqwit, FCEUX, Nestopia, Blargg…). This format lets you supply your
own, so your tiles preview in exactly the colors your target emulator shows.

**Providing your palette:** drop a standard 192-byte ``.pal`` file (64 entries ×
R,G,B — what emulators load) named to match this plugin next to it. Activated as
``nes-custom.py``, it reads ``nes-custom.pal`` from the same folder. With no such
file it falls back to the baked default below, so the format always works.

Like the built-in indexed codec, decode is a table lookup and encode is
nearest-entry by RGB distance (the master palette has duplicate colors, so it has
no exact inverse). It appears in the palette format picker like any preset.
"""

from pathlib import Path

from celpix.core.palette import MISSING_COLOR, Palette
from celpix.plugins import FormatInfo

# Fallback master palette (0xRRGGBB), used when no companion .pal is found — one
# common NES rendering, so the example is self-contained. Replace it by dropping
# your own .pal beside the plugin rather than editing these numbers.
_DEFAULT_MASTER = [
    0x5D5D5D, 0x00218E, 0x120BAF, 0x3900A6, 0x5B0078, 0x6C0033, 0x670500, 0x4E1C00,
    0x293100, 0x044400, 0x004D00, 0x004809, 0x00384F, 0x000000, 0x000000, 0x000000,
    0xA8A8A8, 0x0F54ED, 0x3E33FF, 0x761BFF, 0xA510CE, 0xBD156D, 0xB7290D, 0x934800,
    0x5E6A00, 0x288500, 0x019200, 0x008C30, 0x007595, 0x000000, 0x000000, 0x000000,
    0xFFFFFF, 0x59A7FF, 0x8F83FF, 0xCB67FF, 0xFC5BFF, 0xFF61C1, 0xFF7757, 0xEA990A,
    0xB1BE00, 0x76DA00, 0x48E823, 0x32E17F, 0x38C9EC, 0x454545, 0x000000, 0x000000,
    0xFFFFFF, 0xB9DAFF, 0xD0CBFF, 0xEABFFF, 0xFEBAFF, 0xFFBCE5, 0xFFC6B8, 0xF6D594,
    0xDFE482, 0xC6F087, 0xB2F6A1, 0xA8F3CA, 0xAAE9F7, 0xB0B0B0, 0x000000, 0x000000,
]

_MASTER_COLORS = 64  # NES master palette size; 512-entry (emphasis) .pal files
_PAL_BYTES = _MASTER_COLORS * 3  # are read as their base 64 only.


def _load_master_palette():
    """The 64 master colors: from the companion ``.pal`` if present and valid,
    else the baked default. Called once at register() time — the master palette
    is fixed for the session, so re-reading the file per decode would be waste."""
    try:
        raw = Path(__file__).with_suffix(".pal").read_bytes()
    except OSError:
        return list(_DEFAULT_MASTER)
    if len(raw) < _PAL_BYTES:  # truncated/not a .pal — don't half-apply it
        return list(_DEFAULT_MASTER)
    return [(raw[i * 3] << 16) | (raw[i * 3 + 1] << 8) | raw[i * 3 + 2]
            for i in range(_MASTER_COLORS)]


class NesCustomPalette:
    """Indexed NES palette whose 64-color master table the user supplies."""

    info = FormatInfo(id="format.palette.nes-custom", name="NES/Famicom (Custom)")

    def __init__(self):
        # Render as opaque ARGB; the master table is built once at load.
        self._table = [0xFF000000 | c for c in _load_master_palette()]

    def decode(self, data, ctx):
        n = len(self._table)
        return Palette([self._table[b] if b < n else MISSING_COLOR for b in data])

    def encode(self, palette, ctx):
        return bytes(self._nearest(argb) for argb in palette.colors)

    def bytes_per_entry(self):
        return 1  # one index byte per entry — how Celpix sizes palette reads

    def _nearest(self, argb):
        r, g, b = (argb >> 16) & 0xFF, (argb >> 8) & 0xFF, argb & 0xFF
        best_i, best_d = 0, None
        for i, c in enumerate(self._table):
            d = (abs(r - ((c >> 16) & 0xFF)) + abs(g - ((c >> 8) & 0xFF))
                 + abs(b - (c & 0xFF)))
            if best_d is None or d < best_d:
                best_i, best_d = i, d
        return best_i


def register(registry):
    registry.register_format(NesCustomPalette())
