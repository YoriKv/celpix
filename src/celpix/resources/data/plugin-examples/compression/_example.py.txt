"""Example compression scheme. Files starting with "_" are ignored — copy this
file, drop the underscore from the name, edit it, and press F5 in Celpix to
load it (code plugins ask for confirmation the first time, since they run with
the app's privileges).

A compression scheme is a Decompress/Compress *pair*: both halves live in one
file because they are two ends of one format. Decompress runs as a pipeline
layer *before* the pixel codec — it turns a packed byte run into the raw bytes
the pixel format then interprets; Compress is its inverse, run on save so edits
write back packed. A view-only format may ship Decompress alone (drop the
Compress class and its register call); such a format opens read-only.

This toy scheme is byte run-length encoding: the stream is a series of
``(count, value)`` pairs, ``count`` in 1..255, terminated by a ``count`` of 0.
Real schemes (LZ, Konami RLE) are more involved, but the plumbing is identical.
"""

from celpix.core.context import (
    KEY_COMPRESSED_SIZE,
    KEY_DECOMPRESS_COMPLETE,
    PipelineContext,
)
from celpix.core.errors import Stage
from celpix.plugins.base import PluginInfo


class ExampleRleDecompress:
    # stage marks this the decode half; the id doubles as its stable identity.
    info = PluginInfo(
        id="decompress.example-rle",
        name="Example byte RLE",
        stage=Stage.DECOMPRESS,
    )

    def decompress(self, data: bytes, ctx: PipelineContext) -> bytes:
        """Expand ``(count, value)`` pairs until the 0-count terminator.

        Celpix hands over the visible byte window, which may end mid-stream, so
        decode defensively and report two things back through ``ctx``:

        - ``KEY_COMPRESSED_SIZE`` — bytes consumed, i.e. the packed structure's
          true length; a save-back must fit this slot.
        - ``KEY_DECOMPRESS_COMPLETE`` — whether the terminator was reached
          inside the buffer (False = the window cut the stream short).
        """
        out = bytearray()
        i, n = 0, len(data)
        complete = False
        while i < n:
            count = data[i]
            if count == 0:  # terminator
                i += 1
                complete = True
                break
            if i + 1 >= n:  # window ended before the value byte
                break
            out += bytes([data[i + 1]]) * count
            i += 2
        ctx.set(KEY_COMPRESSED_SIZE, i)
        ctx.set(KEY_DECOMPRESS_COMPLETE, complete)
        return bytes(out)


class ExampleRleCompress:
    info = PluginInfo(
        id="compress.example-rle",
        name="Example byte RLE",
        stage=Stage.COMPRESS,
    )

    def compress(self, data: bytes, ctx: PipelineContext) -> bytes:
        """Pack ``data`` into ``(count, value)`` pairs — the inverse of decode.

        Runs longer than 255 split across successive pairs; the stream ends with
        a 0-count terminator so decode knows where it stops.
        """
        out = bytearray()
        i, n = 0, len(data)
        while i < n:
            value = data[i]
            run = 1
            while i + run < n and data[i + run] == value and run < 255:
                run += 1
            out.append(run)
            out.append(value)
            i += run
        out.append(0)  # terminator
        return bytes(out)


def register(registry):
    registry.register(ExampleRleDecompress())
    registry.register(ExampleRleCompress())
